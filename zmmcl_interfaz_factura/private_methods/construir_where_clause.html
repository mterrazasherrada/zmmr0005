METHOD construir_where_clause.

  "/ Validar si tiene datos de imputación, ref. pestaña imputación
  IF it_pep[] IS NOT INITIAL OR it_kostl[] IS NOT INITIAL OR
     it_aufnr[] IS NOT INITIAL OR it_sakto[] IS NOT INITIAL OR it_prctr[] IS NOT INITIAL
     OR it_gsber[] IS NOT INITIAL.

    CONCATENATE 'c~vgabe IN gr_tipos_fac AND a~bukrs in it_bukrs'
                    'AND a~ekorg IN it_ekorg AND a~ekgrp IN it_ekgrp AND a~lifnr'
                    'IN it_lifnr AND a~bedat IN it_bedat AND a~ebeln IN it_ebeln'
                    'AND d~loekz NE ''L'' AND ( e~zekkn = c~zekkn or c~zekkn = space )'
                    'AND d~konnr IN it_konnr'
                    'AND ( a~frgrl = '' )' "solo documentos liberados
                    'AND ( a~loekz = '' )' "solo documentos no eliminados'
             INTO ep_where_clause SEPARATED BY space.

  ELSE.

    CONCATENATE 'c~vgabe IN gr_tipos_fac AND a~bukrs IN it_bukrs'
            'AND a~ekorg IN it_ekorg AND a~ekgrp IN it_ekgrp AND a~lifnr'
            'IN it_lifnr AND a~bedat IN it_bedat AND a~ebeln IN it_ebeln'
            'AND d~loekz NE ''L''' 'AND d~konnr IN it_konnr'
            'AND ( a~frgrl = '' )' "solo documentos liberados
            'AND ( a~loekz = '' )' "solo documentos no eliminados'
     INTO ep_where_clause SEPARATED BY space.

  ENDIF.

  IF it_pep[] IS NOT INITIAL.
    CONCATENATE ep_where_clause 'AND e~ps_psp_pnr IN it_pep'
        INTO ep_where_clause SEPARATED BY space.
  ENDIF.

  IF it_kostl[] IS NOT INITIAL.
    CONCATENATE ep_where_clause 'AND e~kostl IN it_kostl'
        INTO ep_where_clause SEPARATED BY space.
  ENDIF.

  IF it_aufnr[] IS NOT INITIAL.
    CONCATENATE ep_where_clause 'AND e~aufnr IN it_aufnr'
        INTO ep_where_clause SEPARATED BY space.
  ENDIF.

  IF it_sakto[] IS NOT INITIAL.
    CONCATENATE ep_where_clause 'AND e~sakto IN it_sakto'
        INTO ep_where_clause SEPARATED BY space.
  ENDIF.

  IF it_prctr[] IS NOT INITIAL.
    CONCATENATE ep_where_clause 'AND e~prctr IN it_prctr'
        INTO ep_where_clause SEPARATED BY space.
  ENDIF.

  IF it_gsber[] IS NOT INITIAL.
    CONCATENATE ep_where_clause 'AND e~gsber IN it_gsber'
        INTO ep_where_clause SEPARATED BY space.
  ENDIF.

  "/ Opción 2.- Confirmar carga de facturas
  IF GV_ETAPA = '02' AND me->it_user[] IS NOT INITIAL.
    CONCATENATE ep_where_clause 'AND f~usuario IN it_user'
       INTO ep_where_clause SEPARATED BY space.
  ENDIF.

  "/ Opción 3.- Reporte de estado de facturas
  IF GV_ETAPA = '03' AND ( me->it_status[] IS NOT INITIAL OR me->it_user[] IS NOT INITIAL ).
    IF me->it_status[] IS NOT INITIAL.
      CONCATENATE ep_where_clause 'AND f~estado IN it_status'
         INTO ep_where_clause SEPARATED BY space.
    ENDIF.
    IF me->it_user[] IS NOT INITIAL.
      CONCATENATE ep_where_clause 'AND f~usuario IN it_user'
         INTO ep_where_clause SEPARATED BY space.
    ENDIF.
  ENDIF.

ENDMETHOD.
