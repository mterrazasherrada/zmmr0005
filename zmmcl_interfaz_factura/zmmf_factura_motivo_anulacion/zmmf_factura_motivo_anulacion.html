Description: Motivo de anulacion facturas
FUNCTION zmmf_factura_motivo_anulacion .
*"----------------------------------------------------------------------
*"*"Interfase local
*" IMPORTING
*" VALUE(TITEL) LIKE TRM060-TEXT OPTIONAL
*" VALUE(START_COLUMN) DEFAULT 25
*" VALUE(START_ROW) DEFAULT 6
*" CHANGING
*" VALUE(EW_MOTIVO_ANUL) LIKE ZMMS_FACTURA_MOTIVO_ANULACION
*" STRUCTURE ZMMS_FACTURA_MOTIVO_ANULACION OPTIONAL
*"----------------------------------------------------------------------

* Global data declarations

  DATA: x1 TYPE i VALUE 5,
        y1 TYPE i VALUE 5,
        x2 TYPE i VALUE 19,
        y2 TYPE i VALUE 15.

  x1 = start_column.
  x2 = x1       + 80.
  y1 = start_row.
  y2 = start_row + 2.

  gw_motivo_anul = ew_motivo_anul.
  gv_titel      = titel.

  CALL SCREEN 300 STARTING AT x1 y1 ENDING AT x2 y2.
*
  CASE ok_code.
    WHEN 'ENTE'.
      ew_motivo_anul = gw_motivo_anul.
    WHEN 'CANC'.
      CLEAR gw_motivo_anul.
    WHEN OTHERS.
      CLEAR gw_motivo_anul.
  ENDCASE.

ENDFUNCTION.
