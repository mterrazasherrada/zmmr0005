METHOD toolbar.

  DATA: ls_toolbar LIKE LINE OF e_object->mt_toolbar.

  DELETE e_object->mt_toolbar
      WHERE function = '&DETAIL'           OR function = '&MB_VIEW'
         OR function = '&PRINT_BACK'
         OR function = '&GRAPH'            OR function = '&INFO'         OR function = '&REFRESH'
         OR function = '&LOCAL&CUT'        OR function = '&LOCAL©'   OR function = '&LOCAL&PASTE'
         OR function = '&LOCAL&UNDO'       OR function = '&LOCAL&APPEND' OR function = '&LOCAL&INSERT_ROW'
         OR function = '&LOCAL&DELETE_ROW' OR function = '&LOCAL©_ROW' .

*  "/ Seleccionar disposición...
*  ls_toolbar-function  = '&OAD'.
*  ls_toolbar-butn_type = 0.
*  ls_toolbar-icon      = icon_alv_variant_choose.
*  APPEND ls_toolbar TO e_object->mt_toolbar.
*
*  "/ Grabar disposición...
*  ls_toolbar-function  = '&AVE'.
*  ls_toolbar-butn_type = 0.
*  ls_toolbar-icon      = icon_alv_variant_save.
*  APPEND ls_toolbar TO e_object->mt_toolbar.

  "/ Seleccionar todo
  ls_toolbar-function  = 'ALL'.
  ls_toolbar-butn_type = 0.
  ls_toolbar-icon      = icon_select_all.
  APPEND ls_toolbar TO e_object->mt_toolbar.

  "/ Deseleccionar todo
  ls_toolbar-function  = 'NONE'.
  ls_toolbar-butn_type = 0.
  ls_toolbar-icon      = icon_deselect_all.
  APPEND ls_toolbar TO e_object->mt_toolbar.

  "/ Refrescar
  ls_toolbar-function  = 'REFRESCAR'.
  ls_toolbar-butn_type = 0.
  ls_toolbar-icon      = icon_refresh.
  APPEND ls_toolbar TO e_object->mt_toolbar.

ENDMETHOD.
