METHOD get_imputaciones.

  CHECK et_itab[] IS NOT INITIAL.
  "/ Leer las imputaciones
  SELECT ebeln                "/ Doc
         ebelp                "/ Pos
         zekkn                "/ Nro de imputación
         prctr                "/ Cebe
         ps_psp_pnr           "/ PEP
         kostl                "/ Ceco
         aufnr                "/ Orden
         sakto                "/ Cuenta
         gsber
  INTO TABLE gt_ekkn
  FROM ekkn
  FOR ALL ENTRIES IN et_itab
  WHERE ebeln = et_itab-ebeln
    AND ebelp = et_itab-ebelp
    AND ( ps_psp_pnr IS NOT NULL "/ PEP
          OR kostl IS NOT NULL    "/ Ceco
          OR prctr IS NOT NULL    "/ Cebe
          OR aufnr IS NOT NULL    "/ Orden
          OR sakto IS NOT NULL    "/ Cuenta
    ).
  IF gt_ekkn IS NOT INITIAL.
    SELECT gsber gtext
      FROM tgsbt
      INTO TABLE gt_tgsbt
   FOR ALL ENTRIES IN gt_ekkn
   WHERE gsber = gt_ekkn-gsber
      AND spras = sy-langu.
    ENDIF.

    IF gv_etapa = '02'."Confirmar factura
      "/ Datos de cabecera de hoja de entr.servicios & Líneas del paquete de serv.
      SELECT a~ebeln a~ebelp a~packno b~extrow
        INTO TABLE gt_essr
        FROM essr AS a
        INNER JOIN esll AS b
        ON a~packno = b~packno
        FOR ALL ENTRIES IN et_itab
     WHERE ebeln = et_itab-ebeln
       AND ebelp = et_itab-ebelp
       AND kzabn = 'X'
       AND knttp = et_itab-knttp.
    ENDIF.
  ENDMETHOD.
